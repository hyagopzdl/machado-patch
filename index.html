function shuffle(arr) {

  const a = [...arr];

  for (let i = a.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [a[i], a[j]] = [a[j], a[i]];

  }

  return a;

}


function generateRoundRobin(teamIds) {

  let arr = [...teamIds];

  if (arr.length % 2 !== 0) arr.push(null);

  const n = arr.length;

  const rounds = [];

  for (let r = 0; r < n - 1; r++) {

    const roundMatches = [];

    for (let i = 0; i < n / 2; i++) {

      const home = arr[i], away = arr[n - 1 - i];

      if (home !== null && away !== null) {

        roundMatches.push(r % 2 === 0 ? { home, away } : { home: away, away: home });

      }

    }

    rounds.push(roundMatches);

    const fixed = arr[0];

    const rest = arr.slice(1);

    rest.unshift(rest.pop());

    arr = [fixed, ...rest];

  }

  return rounds;

}


function generateKnockoutRound(teamIds) {

  const s = shuffle(teamIds);

  const matches = [];

  for (let i = 0; i < s.length; i += 2) {

    if (i + 1 < s.length) matches.push({ home: s[i], away: s[i + 1], bye: false });

    else matches.push({ home: s[i], away: null, bye: true });

  }

  return matches;

}


function splitGroups(teamIds, numGroups) {

  const s = shuffle(teamIds);

  const groups = Array.from({ length: numGroups }, () => []);

  s.forEach((id, i) => groups[i % numGroups].push(id));

  return groups;

}


function computeStandings(teamIds, matches) {

  const table = {};

  teamIds.forEach((id) => {

    table[id] = { id, pj: 0, v: 0, e: 0, d: 0, gp: 0, gc: 0, sg: 0, pts: 0 };

  });

  matches.forEach((m) => {

    if (!m.played || m.bye) return;

    const h = table[m.homeId], a = table[m.awayId];

    if (!h || !a) return;

    h.pj++; a.pj++;

    h.gp += m.homeScore; h.gc += m.awayScore;

    a.gp += m.awayScore; a.gc += m.homeScore;

    if (m.homeScore > m.awayScore) { h.v++; a.d++; h.pts += 3; }

    else if (m.homeScore < m.awayScore) { a.v++; h.d++; a.pts += 3; }

    else { h.e++; a.e++; h.pts += 1; a.pts += 1; }

  });

  Object.values(table).forEach((t) => (t.sg = t.gp - t.gc));

  return Object.values(table).sort(

    (x, y) => y.pts - x.pts || y.sg - x.sg || y.gp - x.gp

  );

}


function matchWinner(m) {

  if (m.bye) return m.home;

  if (!m.played) return null;

  if (m.homeScore > m.awayScore) return m.homeId;

  if (m.awayScore > m.homeScore) return m.awayId;

  return m.penaltyWinner || null;

}

function getDb() {

  if (db) return db;

  if (window.firebase && window.__FIREBASE_CONFIG__ && window.__FIREBASE_CONFIG__.apiKey) {

    if (!window.firebase.apps.length) window.firebase.initializeApp(window.__FIREBASE_CONFIG__);

    db = window.firebase.database();

  }

  return db;

}

function dbSet(path, value) {

  const d = getDb();

  if (!d) return Promise.resolve();

  return d.ref("pes/" + path).set(value === undefined ? null : value).catch((e) => console.error("firebase set failed", path, e));

}

function dbSubscribe(path, cb) {

  const d = getDb();

  if (!d) { cb(null); return () => {}; }

  const ref = d.ref("pes/" + path);

  const handler = (snap) => cb(snap.val());

  ref.on("value", handler);

  return () => ref.off("value", handler);

}

function KnockoutView({ tournament, teamById, onOpenMatch, advanceKnockout }) {

  const knockoutMatches = tournament.matches.filter((m) => m.stage === "knockout");

  const rounds = [...new Set(knockoutMatches.map((m) => m.round))].sort((a, b) => a - b);

  const maxRound = Math.max(...rounds);

  const roundLabel = (r, total) => {

    const remaining = total - r + 1;

    const teamsInRound = knockoutMatches.filter((m) => m.round === r).length * 2;

    if (teamsInRound <= 2) return "FINAL";

    if (teamsInRound <= 4) return "SEMIFINAL";

    if (teamsInRound <= 8) return "QUARTAS DE FINAL";

    return `FASE DE ${teamsInRound}`;

  };

  const currentRoundMatches = knockoutMatches.filter((m) => m.round === maxRound);

  const currentComplete = currentRoundMatches.every((m) => m.played);

  const winnersCount = currentComplete ? currentRoundMatches.length : null;


  return (

    <div>

      {rounds.map((r) => (

        <div key={r} style={{ marginBottom: 16 }}>

          <div style={{ fontSize: 12, color: "#6B7A6F", marginBottom: 6, fontWeight: 700, letterSpacing: 1 }}>{roundLabel(r, maxRound)}</div>

          {knockoutMatches.filter((m) => m.round === r).map((m) => (

            <div key={m.id}>

              <MatchRow match={m} teamById={teamById} onOpen={onOpenMatch} />

              {m.played && !m.bye && m.homeScore === m.awayScore && m.penaltyWinner && (

                <div style={{ fontSize: 11.5, color: "#9AAA9F", marginTop: -4, marginBottom: 8, paddingLeft: 4 }}>

                  Pênaltis: {teamById(m.penaltyWinner)?.name} venceu

                </div>

              )}

            </div>

          ))}

        </div>

      ))}

      {currentComplete && winnersCount > 1 && (

        <button className="tapbtn" onClick={advanceKnockout} style={{ ...bigBtn, ...btnPrimary }}>Gerar próxima fase</button>

      )}

      {currentComplete && winnersCount === 1 && (

        <button className="tapbtn" onClick={advanceKnockout} style={{ ...bigBtn, ...btnPrimary }}>🏆 Coroar campeão</button>

      )}

    </div>

  );

}

      )}


      <button className="tapbtn" disabled={!canCreate} onClick={onCreate} style={{ ...bigBtn, ...btnPrimary, opacity: canCreate ? 1 : 0.5 }}>

        Iniciar campeonato

      </button>

    </Modal>

  );

}


// ============ MATCH MODAL ============

function MatchModal({ match, teamById, scoreDraft, setScoreDraft, onClose, onSave, addScorer, removeScorer }) {

  const home = teamById(match.homeId), away = teamById(match.awayId);

  const isDraw = scoreDraft.home !== "" && scoreDraft.away !== "" && parseInt(scoreDraft.home) === parseInt(scoreDraft.away);

  const needsPenalties = match.stage === "knockout" && isDraw;


  return (

    <Modal title="Resultado da partida" onClose={onClose}>

      <div style={{ display: "flex", alignItems: "center", justifyContent: "center", gap: 16, marginBottom: 18 }}>

        <div style={{ textAlign: "center", flex: 1 }}>

          <div style={{ fontWeight: 700, fontSize: 13.5, marginBottom: 8 }}>{home?.name}</div>

          <input type="number" min={0} style={{ ...inputStyle, textAlign: "center", fontSize: 22, fontWeight: 800, padding: "10px 0" }}

            value={scoreDraft.home} onChange={(e) => setScoreDraft({ ...scoreDraft, home: e.target.value })} />

        </div>

        <span style={{ fontSize: 18, color: "#6B7A6F", marginTop: 20 }}>×</span>

        <div style={{ textAlign: "center", flex: 1 }}>

          <div style={{ fontWeight: 700, fontSize: 13.5, marginBottom: 8 }}>{away?.name}</div>

          <input type="number" min={0} style={{ ...inputStyle, textAlign: "center", fontSize: 22, fontWeight: 800, padding: "10px 0" }}

            value={scoreDraft.away} onChange={(e) => setScoreDraft({ ...scoreDraft, away: e.target.value })} />

        </div>

      </div>


      {needsPenalties && (

        <div style={{ marginBottom: 16 }}>

          <label style={lbl}>Empate — vencedor nos pênaltis</label>

          <div style={{ display: "flex", gap: 8 }}>

            <span onClick={() => setScoreDraft({ ...scoreDraft, penaltyWinner: match.homeId })} style={{ ...chip(scoreDraft.penaltyWinner === match.homeId), flex: 1, textAlign: "center" }}>{home?.name}</span>

            <span onClick={() => setScoreDraft({ ...scoreDraft, penaltyWinner: match.awayId })} style={{ ...chip(scoreDraft.penaltyWinner === match.awayId), flex: 1, textAlign: "center" }}>{away?.name}</span>

          </div>

        </div>

      )}


      <label style={lbl}>Gols marcados (opcional)</label>

      <div style={{ display: "flex", gap: 8, marginBottom: 10 }}>

        <PlayerScorerPicker team={home} onPick={(pid, pname) => addScorer(match.homeId, pid, pname)} />

        <PlayerScorerPicker team={away} onPick={(pid, pname) => addScorer(match.awayId, pid, pname)} />

      </div>

      {scoreDraft.scorers.length > 0 && (

        <div style={{ marginBottom: 14 }}>

          {scoreDraft.scorers.map((s, i) => (

            <div key={i} style={{ display: "flex", alignItems: "center", justifyContent: "space-between", fontSize: 13, padding: "5px 0", borderBottom: i < scoreDraft.scorers.length - 1 ? "1px solid #1E3528" : "none" }}>

              <span><Target size={12} style={{ verticalAlign: -1, marginRight: 5 }} color="#D4AF37" />{s.playerName}</span>

              <button onClick={() => removeScorer(i)} style={{ background: "none", border: "none", color: "#C1443D", cursor: "pointer" }}><X size={14} /></button>

            </div>

          ))}

        </div>

      )}


      <button className="tapbtn" onClick={onSave} style={{ ...bigBtn, ...btnPrimary }}>Salvar resultado</button>

    </Modal>

  );

}


function PlayerScorerPicker({ team, onPick }) {

  const [open, setOpen] = useState(false);

  if (!team) return <div style={{ flex: 1 }} />;

  return (

    <div style={{ position: "relative", flex: 1 }}>

      <button onClick={() => setOpen((o) => !o)} style={{

        width: "100%", padding: "9px 10px", borderRadius: 8, background: "#0E1E16", border: "1px solid #24402F", color: "#9AAA9F", fontSize: 12, cursor: "pointer", display: "flex", alignItems: "center", justifyContent: "space-between",

      }}>

        + Gol {team.name} <ChevronDown size={13} />

      </button>

      {open && (

        <div style={{ position: "absolute", top: "105%", left: 0, right: 0, background: "#182B20", border: "1px solid #2F4A38", borderRadius: 8, zIndex: 5, maxHeight: 160, overflowY: "auto" }}>

          {team.players.length === 0 && <div style={{ padding: 10, fontSize: 12, color: "#6B7A6F" }}>Sem jogadores</div>}

          {team.players.map((p) => (

            <div key={p.id} onClick={() => { onPick(p.id, p.name); setOpen(false); }} style={{ padding: "8px 10px", fontSize: 12.5, cursor: "pointer", borderBottom: "1px solid #1E3528" }}>

              {p.name}

            </div>

          ))}

        </div>

      )}

    </div>

  );

}

function TransferModal({ pick, teams, onClose, onConfirm }) {

  const { player, fromTeamId } = pick;

  const from = teams.find((t) => t.id === fromTeamId);

  const [dest, setDest] = useState(null);

  return (

    <Modal title="Transferir jogador" onClose={onClose}>

      <div style={{ ...panelStyle, marginBottom: 16 }}>

        <div style={{ fontWeight: 700, fontSize: 15 }}>{player.name}</div>

        <div style={{ fontSize: 12, color: "#9AAA9F" }}>{from?.name} · OVR {player.ovr} · {POS_LABELS[player.pos]}</div>

      </div>

      <label style={lbl}>Destino</label>

      <div style={{ display: "flex", flexDirection: "column", gap: 6, marginBottom: 18 }}>

        {teams.filter((t) => t.id !== fromTeamId).map((t) => (

          <button key={t.id} onClick={() => setDest(t.id)} style={{

            display: "flex", alignItems: "center", gap: 10, padding: 10, borderRadius: 10, cursor: "pointer",

            background: dest === t.id ? "rgba(212,175,55,0.12)" : "#0E1E16", border: dest === t.id ? "1px solid #D4AF37" : "1px solid #24402F", color: "#EDEFE7",

          }}>

            <span style={{ width: 8, height: 8, borderRadius: 3, background: t.color }} />

            <span style={{ flex: 1, textAlign: "left", fontSize: 14 }}>{t.name}</span>

            {dest === t.id && <Check size={16} color="#D4AF37" />}

          </button>

        ))}

        {teams.length <= 1 && <div style={{ fontSize: 12.5, color: "#6B7A6F" }}>Crie outro time para transferir.</div>}

      </div>

      <button className="tapbtn" disabled={!dest} onClick={() => onConfirm(player, fromTeamId, dest)} style={{ ...bigBtn, ...btnPrimary, opacity: dest ? 1 : 0.5 }}>

        Confirmar transferência

      </button>

    </Modal>

  );

}
